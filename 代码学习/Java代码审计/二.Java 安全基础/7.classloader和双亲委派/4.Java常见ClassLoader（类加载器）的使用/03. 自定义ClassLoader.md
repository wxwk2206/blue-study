# 0x05 例子-调用MaliciousClass类执行命令
```java
package ClassLoader.CustomClassLoaderAddClassFile;

public class ByteClassLoader extends ClassLoader {
    private String loadClassName = "";
    private byte[] loadClassBytes = {};

    public ByteClassLoader(String loadClassName, byte[] loadClassBytes) {
        this.loadClassName = loadClassName;
        this.loadClassBytes = loadClassBytes;
    }

    @Override
    public Class<?> findClass(String name) throws ClassNotFoundException {
        if (name.equals(this.loadClassName)) {
            // 调用JVM的native方法定义我们给的类
            return defineClass(this.loadClassName, this.loadClassBytes, 0, this.loadClassBytes.length);
        }
        return super.findClass(name);
    }
}
```
