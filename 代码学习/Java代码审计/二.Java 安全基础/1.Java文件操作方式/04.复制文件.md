## 三、核心复制方法
### （一）基础 IO 流实现（JDK 原生）
#### 1. FileInputStream 与 FileOutputStream 配合
+ 原理：通过字节输入流读取源文件，字节输出流写入目标文件，手动控制缓冲区
+ 代码示例：

```java
package CopyTest;
import java.io.File;
import java.io.FileInputStream;
import java.io.FileOutputStream;
import java.io.IOException;

public class CopyTest1 {
    public static void main(String[] args) {
        File sourceFile = new File("1.txt");
        copyFileUsingStream(sourceFile);
    }

    private static void copyFileUsingStream(File sourceFile) {
        FileInputStream fis = null;
        FileOutputStream fos = null;
        String targetPath = "copy-" + sourceFile.getName();
        try {
            fis = new FileInputStream(sourceFile);
            fos = new FileOutputStream(targetPath);
            int len = 0;
            byte[] bytes = new byte[1024]; // 1KB缓冲区
            while ((len = fis.read(bytes)) != -1) {
                fos.write(bytes, 0, len); // 写入读取到的有效字节
            }
        } catch (IOException e) {
            e.printStackTrace();
        } finally {
            try {
                if (fis != null) fis.close();
                if (fos != null) fos.close();
            } catch (IOException e) {
                e.printStackTrace();
            }
        }
    }
}
```

#### 2. FileChannel（复制速度最快）
+ 原理：基于 NIO 通道的直接数据传输，减少中间缓冲区拷贝，效率最高
+ 代码示例：

```java
package CopyTest;
import java.io.File;
import java.io.FileInputStream;
import java.io.FileOutputStream;
import java.io.IOException;
import java.nio.channels.FileChannel;

public class CopyTest2 {
    public static void main(String[] args) {
        File sourceFile = new File("./1.txt");
        copyFileChannel(sourceFile);
    }

    private static void copyFileChannel(File sourceFile) {
        FileInputStream fis = null;
        FileOutputStream fos = null;
        String targetPath = "copy-" + sourceFile.getName();
        try {
            fis = new FileInputStream(sourceFile);
            fos = new FileOutputStream(targetPath);
            FileChannel fisChannel = fis.getChannel();
            //  获取文件通道（关联到文件描述符，直接对接操作系统）
            FileChannel fosChannel = fos.getChannel();
            // 直接通道传输数据
            fisChannel.transferTo(0, fisChannel.size(), fosChannel);
        } catch (IOException e) {
            e.printStackTrace();
        } finally {
            try {
                if (fis != null) fis.close();
                if (fos != null) fos.close();
            } catch (IOException e) {
                e.printStackTrace();
            }
        }
    }
}
```

+ 执行结果：根目录生成 `copy-1.txt`，内容为 `111111`

#### 3. Files.copy ()（JDK 7+ 推荐）
+ 特点：JDK 7 引入的 `java.nio.file.Files` 工具类，简化复制逻辑，底层使用文件系统提供程序
+ 代码示例：

```java
package CopyTest;
import java.io.File;
import java.io.IOException;
import java.nio.file.Files;

public class CopyTest3 {
    public static void main(String[] args) throws IOException {
        File sourceFile = new File("./1.txt");
        copy(sourceFile);
    }

    private static void copy(File sourceFile) {
        String targetPath = "copy-" + sourceFile.getName();
        File targetFile = new File(targetPath);
        try {
            Files.copy(sourceFile.toPath(), targetFile.toPath());
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
}
```

+ 执行结果：根目录生成 `copy-1.txt`，内容为 `111111`

#### 1. StreamUtils
+ 用途：Spring 提供的流操作工具类，简化流复制逻辑
+ 代码示例：

```java
package CopyTest;
import org.springframework.util.StreamUtils;
import java.io.File;
import java.io.FileInputStream;
import java.io.FileOutputStream;
import java.io.IOException;

public class CopyTest4 {
    public static void main(String[] args) {
        File sourceFile = new File("./1.txt");
        copy(sourceFile);
    }

    private static void copy(File sourceFile) {
        FileInputStream fis = null;
        FileOutputStream fos = null;
        String targetPath = "copy-" + sourceFile.getName();
        try {
            fis = new FileInputStream(sourceFile);
            fos = new FileOutputStream(targetPath);
            StreamUtils.copy(fis, fos); // 直接复制流
        } catch (IOException e) {
            e.printStackTrace();
        } finally {
            try {
                if (fis != null) fis.close();
                if (fos != null) fos.close();
            } catch (IOException e) {
                e.printStackTrace();
            }
        }
    }
}
```

+ 执行结果：根目录生成 `copy-1.txt`，内容为 `111111`

+ 执行结果：根目录生成 `copy-1.txt`，内容为 `111111`

#### 3. IOUtils 流复制
##### （1）普通文件复制
```java
package CopyTest;

import org.apache.commons.io.IOUtils;
import java.io.FileInputStream;
import java.io.FileOutputStream;
import java.io.IOException;
import java.io.InputStream;
import java.io.OutputStream;

public class CopyTest11 {
    public static void main(String[] args) {
        String sourcePath = "源文件路径.txt"; // 替换为实际路径
        String targetPath = "目标文件路径.txt"; // 替换为实际路径

        try (InputStream is = new FileInputStream(sourcePath);
             OutputStream os = new FileOutputStream(targetPath)) {
            IOUtils.copy(is, os);
            System.out.println("复制成功");
        } catch (IOException e) {
            System.out.println("复制失败：" + e.getMessage());
        }
    }
}
```
