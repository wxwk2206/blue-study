# 1.JDBC 框架概述
Spring JDBC是Spring所提供的持久层技术

它主要目的降低JDBC API的使用难度,以一种更直接、更简洁的方式使用JDBC API

# 2.JdbcTemplate类介绍
```plain
JdbcTemplate类是Spring框架中提供的一个对象,是对原始繁琐的Jdbc API对象的简单封装

在 JdbcTemplate 类中,提供了大量的查询和更新数据库的方法
public int update(String sql):
用于执行新增、修改、删除等语句

public int update(String sql,Object... args):
用于执行新增、修改、删除等语句
args表示需要传入到query中的参数

public void execute(String sql):
可以执行任意SQL,一般用于执行DDL语句

public T execute(String sql,PreparedStatementCallback action):
可以执行任意SQL,一般用于执行DDL语句,action表示执行完SQL语句后,要调用的函数

public T query(String sql, ResultSetExtractor rse):
用于执行查询语句
以ResultSetExtractor作为参数的query方法返回值为Object


public List query(String sql, RowMapper rse):
用于执行查询语句
以ResultSetExtractor作为参数的query方法返回值为Object,使用查询结果需要对其进行强制转型
以RowMapper作为参数的query方法返回值为List
```

```xml
<!-- 第三步 -->
<!-- 通过XML配置JDBC功能 -->
<!-- 目录: ./SpringDemo/src/main/resources/ -->
<!-- 文件名: jdbcConfig.xml -->
<?xml version="1.0" encoding="UTF-8"?>
<beans xmlns="http://www.springframework.org/schema/beans"
  xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
  xmlns:context="http://www.springframework.org/schema/context"
  xsi:schemaLocation="http://www.springframework.org/schema/beans
  https://www.springframework.org/schema/beans/spring-beans.xsd
  http://www.springframework.org/schema/context
  https://www.springframework.org/schema/context/spring-context.xsd">

  <!-- 导入JDBC配置 -->
  <context:property-placeholder location="classpath:config/jdbc.properties"/>

  <!-- 配置数据源 -->
  <bean id="dataSourceTest" class="org.springframework.jdbc.datasource.DriverManagerDataSource">
    <!--数据库驱动-->
    <property name="driverClassName" value="${jdbc.driver}"/>
    <!--连接数据库的url-->
    <property name="url" value="${jdbc.url}"/>
    <!--连接数据库的用户名-->
    <property name="username" value="${jdbc.username}"/>
    <!--连接数据库的密码-->
    <property name="password" value="${jdbc.password}"/>
  </bean>

  <!-- Spring JDBC 模版 -->
  <bean id="jdbcTemplate" class="org.springframework.jdbc.core.JdbcTemplate" lazy-init="false">
    <property name="dataSource" ref="dataSourceTest"/>
  </bean>
</beans>
```

# 5.基础测试例子
```java
// 创建一个测试类JdbcTest进行测试
// 目录: ./SpringDemo/src/test/java/
// 文件名: JdbcTest.java
import org.springframework.jdbc.core.JdbcTemplate;
import org.springframework.context.ApplicationContext;
import org.springframework.context.support.ClassPathXmlApplicationContext;

public class JdbcTest {
    public static void main(String[] args) {
        ApplicationContext obj = new ClassPathXmlApplicationContext("jdbcConfig.xml");

        // 获取数据库连接
        JdbcTemplate jdbcTemplateObject = (JdbcTemplate) obj.getBean("jdbcTemplate");

        // 在表中插入一行
        String name = "小王";
        int age = 18;
        String SQL = "INSERT INTO student (name, age) VALUES (?, ?)";
        jdbcTemplateObject.update(SQL, name, age);
        System.out.println("------在表中插入一行-------");
        System.out.println("INSERT INTO student (name, age) VALUES (" + name + ", " + age + ")");
        System.out.println(" ");

        // 更新表中的一行
        int id2 = 1;
        String name2 = "小陈";
        int age2 = 25;
        String SQL2 = "UPDATE student SET name = ?, age = ? WHERE id = ?";
        jdbcTemplateObject.update(SQL2, new Object[]{name2, age2, id2});
        System.out.println("------更新表中的一行-------");
        System.out.println("UPDATE student SET name = " + name2 + ", age = " + age2 + " WHERE id = " + id2);
        System.out.println(" ");

        // 查询字符串,使用绑定变量的简单查询
        int id3 = 1;
        String SQL3 = "SELECT name FROM student WHERE id = ?";
        String studentName = jdbcTemplateObject.queryForObject(SQL3, new Object[]{id3}, String.class);
        System.out.println("------查询字符串-------");
        System.out.println("student表,id=" + id3 + "的name值为" + studentName);
        System.out.println("SELECT name FROM student WHERE id = " + id3);
        System.out.println(" ");

        // 从表中删除一行
        int id4 = 1;
        String SQL4 = "DELETE FROM student WHERE id = ?";
        jdbcTemplateObject.update(SQL4, new Object[]{id4});
        System.out.println("------从表中删除一行-------");
        System.out.println("DELETE FROM student WHERE id = " + id4);
        System.out.println(" ");

        // 执行DDL语句,删除student所有数据
        // 你可以使用jdbcTemplate中的execute(..)方法来执行任何SQL语句或DDL语句
        String SQL5 = "DELETE FROM student";
        jdbcTemplateObject.execute(SQL5);

        String SQL6 = "ALTER TABLE student AUTO_INCREMENT=1";
        jdbcTemplateObject.execute(SQL6);
        System.out.println("------执行DDL语句-------");
        System.out.println(SQL5);
        System.out.println(SQL6);
        System.out.println(" ");
    }
}
```
