## 一、介绍
在 SpringBoot 中使用 DataSource 十分便捷，框架提供了默认支持的数据源类型，也支持集成第三方数据源，还能通过注解灵活配置多数据源，满足不同项目的数据库连接需求。

## 二、环境搭配
若未搭建 SpringBoot 环境，可参考：Spring Boot中创建Spring Boot项目-详细过程

## 三、默认数据源
### （一）核心说明
1. 支持的数据源类型（定义在 `org.springframework.boot.autoconfigure.jdbc.DataSourceAutoConfiguration` 中）：
    - `org.apache.tomcat.jdbc.pool.DataSource`
    - `com.zaxxer.hikari.HikariDataSource`
    - `org.apache.commons.dbcp.BasicDataSource`
    - `org.apache.commons.dbcp2.BasicDataSource`
2. 生效优先级：Tomcat -> Hikari -> Dbcp -> Dbcp2
3. 版本差异：
    - SpringBoot 1.0：默认数据源为 Tomcat，导入 jdbc、jpa 或 mybatis 依赖即自动引入
    - SpringBoot 2.0+：默认数据源改为 Hikari，导入 jdbc、jpa 或 mybatis 依赖即自动引入

### （三）普通使用
在 `./src/main/java/com/example/demo/controller` 目录创建 `DbTest1Controller.java`：

```java
package com.example.demo.controller;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.boot.autoconfigure.jdbc.DataSourceProperties;
import org.springframework.context.ApplicationContext;
import org.springframework.jdbc.core.JdbcTemplate;
import org.springframework.stereotype.Controller;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.ResponseBody;

import javax.sql.DataSource;
import java.util.List;
import java.util.Map;

@Controller
public class DbTest1Controller {
    @Autowired
    DataSourceProperties dataSourceProperties;
    @Autowired
    ApplicationContext applicationContext;

    @RequestMapping("/DbTest1/index")
    @ResponseBody
    public String index() {
        // 获取默认数据源
        DataSource dataSource = applicationContext.getBean(DataSource.class);
        // 执行 SQL 查询
        JdbcTemplate jdbcTemplate = new JdbcTemplate(dataSource);
        List<?> resultList = jdbcTemplate.queryForList("select * from user");

        // 拼接结果信息
        String s = "";
        s += dataSource + "<br/>";
        s += dataSource.getClass().getName() + "<br/>";
        s += dataSourceProperties + "<br/><br/>";

        // 展示查询数据
        for (Object result : resultList) {
            Map<String, Object> r = (Map<String, Object>) result;
            s += "用户名: " + r.get("User") + " ";
            s += "密码: " + r.get("Password") + "<br/>";
        }
        return s;
    }
}
```

+ 测试访问：`http://127.0.0.1:8081/DbTest1/index`，可查看数据源信息及查询结果

### （四）通过注解配置多数据源
适用于项目中需要连接多个数据库的场景，以下以配置两个数据源为例：

#### 1. 数据库配置（application.yml）
```yaml
spring:
  datasource:
    # 主数据库配置
    data1:
      jdbc-url: jdbc:mysql://192.168.24.135:3306/mysql?useUnicode=true&characterEncoding=UTF-8&serverTimezone=Asia/Shanghai
      username: root
      password: 123456
      driver-class-name: com.mysql.jdbc.Driver
    # 副数据库配置
    data2:
      jdbc-url: jdbc:mysql://192.168.24.135:3306/mysql?useUnicode=true&characterEncoding=UTF-8&serverTimezone=Asia/Shanghai
      username: root
      password: 123456
      driver-class-name: com.mysql.jdbc.Driver
```

注意：SpringBoot 2.0+ 需使用 `jdbc-url` 配置数据库地址，1.5 版本使用 `url`

#### 2. 创建数据源配置类
在 `./src/main/java/com/example/demo` 下创建 `datasource` 目录，分别创建主、副数据源配置类：

##### （1）主数据源：DataSource1.java
```java
package com.example.demo.datasource;

import org.apache.ibatis.session.SqlSessionFactory;
import org.mybatis.spring.SqlSessionFactoryBean;
import org.mybatis.spring.SqlSessionTemplate;
import org.mybatis.spring.annotation.MapperScan;
import org.springframework.beans.factory.annotation.Qualifier;
import org.springframework.boot.jdbc.DataSourceBuilder;
import org.springframework.boot.context.properties.ConfigurationProperties;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.context.annotation.Primary;
import org.springframework.jdbc.datasource.DataSourceTransactionManager;
import javax.sql.DataSource;

@Configuration      //表示当前类为Spring的配置类，Spring启动时会自动扫描并加载配置
@MapperScan(sqlSessionFactoryRef = "data1SqlSessionFactory")
//指定 MyBatis Mapper 接口的扫描规则。sqlSessionFactoryRef = "data1SqlSessionFactory"：
//表示扫描到的 Mapper 接口，会自动关联名称为 data1SqlSessionFactory 的会话工厂
//（即当前类创建的 SqlSessionFactory），意味着这些 Mapper 会使用 data1 数据源进行数据库操作。
public class DataSource1 {// 定义配置类 DataSource1，用于封装 data1 数据源的所有配置逻辑
    @Bean(name = "data1Source")//方法的返回值被注册为Bean，并且给它起名叫data1Source
    @Primary // 标识为主数据源，默认使用
    @ConfigurationProperties(prefix = "spring.datasource.data1")
    //将配置文件（application.yml/application.properties）中 前缀为 spring.datasource.data1
    //的属性，自动绑定到当前方法创建的 DataSource 实例中
    public DataSource dataSource() {
//访问修饰符 返回类型 方法名(参数)
//public DataSource dataSource() {}这个整体叫方法，大括号{}里面的内容叫方法体
        return DataSourceBuilder.create().build();
        //DataSourceBuilder.create() 创建一个 DataSource 构建器
        //build() 构建出 DataSource 实例
        //return 将这个实例作为方法的返回值
    }

    @Bean(name = "data1SqlSessionFactory")// 配置会话工厂
    @Primary    //表示当前为主会话工厂，未指定工厂的Mapper会优先使用该工厂
    public SqlSessionFactory sqlSessionFactory(@Qualifier("data1Source") DataSource ds) throws Exception {
        //@Qualifier("data1Source")：指定注入名称为 data1Source 的 DataSource 实例
        //DataSource ds：注入的 data1 数据源实例，用于给 SqlSessionFactory 绑定数据源
        //throws Exception：声明可能抛出的异常
        SqlSessionFactoryBean bean = new SqlSessionFactoryBean();
        bean.setDataSource(ds);
        //将注入的 data1 数据源（ds）绑定到 SqlSessionFactoryBean，让后续创建的 
        //SqlSession 都使用该数据源进行数据库连接
        return bean.getObject();//创建并返回 SqlSessionFactory 实例
    }

    @Bean(name = "data1SqlSessionTemplate")// 配置会话模板
    @Primary//标识当前事务管理器是主事务管理器
    public SqlSessionTemplate sqlSessionTemplate(@Qualifier("data1SqlSessionFactory") SqlSessionFactory sessionFactory) throws Exception {
    //@Qualifier("data1SqlSessionFactory"):指定注入的SqlSessionFactory是名称为data1SqlSessionFactory 的 Bean  
        return new SqlSessionTemplate(sessionFactory);
    //return new SqlSessionTemplate(sessionFactory);创建并返回 SqlSessionTemplate的实例sessionFactory
    }

    @Bean(name = "data1TransactionManager")   // 配置事务管理器
    @Primary
    public DataSourceTransactionManager transactionManager(@Qualifier("data1Source") DataSource ds) {
        return new DataSourceTransactionManager(ds);
    }
}
```

package com.example.demo.datasource;

import org.apache.ibatis.session.SqlSessionFactory;
import org.mybatis.spring.SqlSessionFactoryBean;
import org.mybatis.spring.SqlSessionTemplate;
import org.mybatis.spring.annotation.MapperScan;
import org.springframework.beans.factory.annotation.Qualifier;
import org.springframework.boot.jdbc.DataSourceBuilder;
import org.springframework.boot.context.properties.ConfigurationProperties;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.jdbc.datasource.DataSourceTransactionManager;

import javax.sql.DataSource;

@Configuration
@MapperScan(sqlSessionFactoryRef = "data2SqlSessionFactory")
public class DataSource2 {
    // 配置副数据源（无 @Primary 注解）
    @Bean(name = "data2Source")
    @ConfigurationProperties(prefix = "spring.datasource.data2")
    public DataSource dataSource() {
        return DataSourceBuilder.create().build();
    }

    // 配置会话工厂
    @Bean(name = "data2SqlSessionFactory")
    public SqlSessionFactory sqlSessionFactory(@Qualifier("data2Source") DataSource ds) throws Exception {
        SqlSessionFactoryBean bean = new SqlSessionFactoryBean();
        bean.setDataSource(ds);
        return bean.getObject();
    }

    // 配置会话模板
    @Bean(name = "data2SqlSessionTemplate")
    public SqlSessionTemplate sqlSessionTemplate(@Qualifier("data2SqlSessionFactory") SqlSessionFactory sessionFactory) throws Exception {
        return new SqlSessionTemplate(sessionFactory);
    }

    // 配置事务管理器
    @Bean(name = "data2TransactionManager")
    public DataSourceTransactionManager transactionManager(@Qualifier("data2Source") DataSource ds) {
        return new DataSourceTransactionManager(ds);
    }
}

#### 3. 创建测试控制器
##### （1）主数据源测试：DbTest2Controller.java
```java
package com.example.demo.controller;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.boot.autoconfigure.jdbc.DataSourceProperties;
import org.springframework.context.ApplicationContext;
import org.springframework.jdbc.core.JdbcTemplate;
import org.springframework.stereotype.Controller;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.ResponseBody;

import javax.annotation.Resource;
import javax.sql.DataSource;
import java.util.List;
import java.util.Map;

@Controller
//标识当前类是 SpringMVC 的控制器，Spring 启动时会将其纳入 IOC 容器管理，用于接收前端 HTTP 请求
public class DbTest2Controller { //公共类数据库测试2控制器
    @Autowired
    DataSourceProperties dataSourceProperties;
    //Spring 自动注入DataSourceProperties类dataSourceProperties实例对象
    @Autowired
    ApplicationContext applicationContext;
     //Spring 自动注入ApplicationContext类的applicationContext实例对象

    @Resource(name = "data1Source")
    //注入 DataSource1 配置类中定义的主数据源 Bean（与 @Bean(name = "data1Source") 对应）
    private DataSource data1Source;
    //注入的主数据源实例，后续数据库操作均基于该数据源的连接

    @RequestMapping("/DbTest2/index")
    //映射 HTTP 请求路径，前端访问 http://127.0.0.1:8081/DbTest2/index 时会触发该方法
    @ResponseBody
    //将方法返回的字符串直接转为HTTP响应体（无需视图解析器跳转页面），前端可直接看到文本形式的结果
    public String index() {
    //公共返回类型String,方法名index
        JdbcTemplate jdbcTemplate = new JdbcTemplate(data1Source);
//创建 JdbcTemplate 对象，使用 data1Source 数据源。赋值给 jdbcTemplate 变量
        List<?> resultList = jdbcTemplate.queryForList("select * from user");
//执行SQL查询，结果存入resultList；List<?>：不确定类型的List

        String s = "";
        s += data1Source + "<br/>";
        s += data1Source.getClass().getName() + "<br/>";
        s += dataSourceProperties + "<br/><br/>";
//data1Source：输出数据源实例（如 HikariDataSource (HikariPool-1)），验证数据源是否创建成功；
//data1Source.getClass().getName()：输出数据源实现类全路径（如 com.zaxxer.hikari.HikariDataSource），确认使用的数据源类型；
//dataSourceProperties：输出数据源配置属性（如用户名、驱动类等），验证配置是否读取正确；
//<br/>：HTML 换行标签，使前端展示结果更整洁

        for (Object result : resultList) {
            Map<String, Object> r = (Map<String, Object>) result;
            s += "用户名: " + r.get("User") + " ";
            s += "密码: " + r.get("Password") + "<br/>";
//遍历 user 表查询结果：
//将列表中的每个元素转为 Map<String, Object>（因为 queryForList 返回的是 “列名 - 列值” 映射）；
//通过 r.get("User")、r.get("Password") 获取 User 列（用户名）和 Password 列（密码）的值，拼接成字符串；
//最终前端会看到所有用户的用户名和密码信息。
        }
        return s;
//将拼接好的 “数据源信息 + 查询结果” 字符串返回给前端，完成响应
    }
}
```

### （三）创建数据源配置类：DataSource3.java
```java
package com.example.demo.datasource;

import com.alibaba.druid.pool.DruidDataSource;
import org.apache.ibatis.session.SqlSessionFactory;
import org.mybatis.spring.SqlSessionFactoryBean;
import org.mybatis.spring.SqlSessionTemplate;
import org.mybatis.spring.annotation.MapperScan;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.beans.factory.annotation.Qualifier;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.core.env.Environment;
import org.springframework.jdbc.datasource.DataSourceTransactionManager;

import javax.sql.DataSource;

@Configuration
@MapperScan(sqlSessionFactoryRef = "data3SqlSessionFactory")
public class DataSource3 {
    @Autowired
    private Environment env;

    // 配置 Druid 数据源
    @Bean(name = "data3Source")
    public DataSource dataSource() {
        DruidDataSource dataSource = new DruidDataSource();
        dataSource.setUrl(env.getProperty("spring.datasource.data3.jdbc-url"));
        dataSource.setUsername(env.getProperty("spring.datasource.data3.username"));
        dataSource.setPassword(env.getProperty("spring.datasource.data3.password"));
        dataSource.setDriverClassName(env.getProperty("spring.datasource.data3.driver-class-name"));
        return dataSource;
    }

    // 配置会话工厂
    @Bean(name = "data3SqlSessionFactory")
    public SqlSessionFactory sqlSessionFactory(@Qualifier("data3Source") DataSource ds) throws Exception {
        SqlSessionFactoryBean bean = new SqlSessionFactoryBean();
        bean.setDataSource(ds);
        return bean.getObject();
    }

    // 配置会话模板
    @Bean(name = "data3SqlSessionTemplate")
    public SqlSessionTemplate sqlSessionTemplate(@Qualifier("data3SqlSessionFactory") SqlSessionFactory sessionFactory) throws Exception {
        return new SqlSessionTemplate(sessionFactory);
    }

    // 配置事务管理器
    @Bean(name = "data3TransactionManager")
    public DataSourceTransactionManager transactionManager(@Qualifier("data3Source") DataSource ds) {
        return new DataSourceTransactionManager(ds);
    }
}
```
